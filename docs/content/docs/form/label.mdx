title: Label
description: Describes a form field with a label, description, and error message (if any). This widget is usually used for custom form fields. All form fields in Forui come with this widget wrapped.
apiReference: https://pub.dev/documentation/forui/latest/forui.widgets.label/
---

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';
import { Widget } from '@/components/demo/widget';
import { CodeSnippet } from '@/components/code-snippet/code-snippet';
import horizontalSnippet from '@/snippets/examples/label/horizontal.json';
import verticalSnippet from '@/snippets/examples/label/vertical.json';
import disabledSnippet from '@/snippets/examples/label/disabled.json';
import errorSnippet from '@/snippets/examples/label/error.json';

<Tabs items={['Preview', 'Code']}>
  <Tab value="Preview">
    <Widget name='label' variant='horizontal' query={{}}/>
  </Tab>
  <Tab value="Code">
    <CodeSnippet snippet={horizontalSnippet} />
  </Tab>
</Tabs>

## CLI

To generate and customize this style:

```shell copy
dart run forui style create label
```

## Usage

### `FLabel(...)`

```dart copy
FLabel(
  axis: Axis.horizontal,
  child: const Placeholder(),
  style: (style) => style.copyWith(...),
  label: const Text('Accept terms and conditions'),
  description: const Text('You agree to our terms and conditions.'),
  error: const Text('Please accept the terms and conditions.'),
  expands: false,
  states: {WidgetState.error},
);
```
