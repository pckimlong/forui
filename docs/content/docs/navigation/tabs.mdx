import { Tabs, Tab } from 'fumadocs-ui/components/tabs';
import { Widget } from '@/components/demo/widget';
import { CodeSnippet } from '@/components/code-snippet/code-snippet';
import defaultSnippet from '@/snippets/examples/tabs/default.json';

<Tabs items={['Preview', 'Code']}>
  <Tab value="Preview">
    <Widget name='tabs' query={{}} height={500}/>
  </Tab>
  <Tab value="Code">
    <CodeSnippet snippet={defaultSnippet} />
  </Tab>
</Tabs>

## CLI

To generate and customize this style:

```shell copy
dart run forui style create tabs
```

## Usage

### `FTabs(...)`

```dart copy
FTabs(
  style: (style) => style.copyWith(...),
  initialIndex: 1,
  onPress: (index) {},
  children: const [
    FTabEntry(label: Text('Account'), child: Placeholder()),
    FTabEntry(label: Text('Password'), child: Placeholder()),
  ],
);
```
